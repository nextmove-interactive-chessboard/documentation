import os
import time
import socket

import chess
import chess.pgn
import board


class MultiGame:
    def __init__(self, encoder, button, rows, cols, 
                 tiles, squares, pixels, userCode):
        self.board = chess.Board()
        self.pickedSquare = None
        self.pinSquare = None
        self.legalMoves = []
        self.previousSensor = {}

        # Colors
        self.colorSource = (100, 0, 200)
        self.colorTarget = (50, 50, 50)
        self.colorCapture = (250, 0, 0)
        self.colorPinner = (250, 0, 0)
        self.animationColor = (255, 0, 255)
        self.colorOff = (0, 0, 0)
        self.colorGreen = (0, 255, 0)
        self.colorRed = (255, 0, 0)

        # Timings
        self.rowDischargeTime = 0.005
        self.rowChargeTime = 0.005
        self.debounceTime = 0.2
        self.blinkInterval = 0.4

        self.encoder = encoder
        self.button = button

        self.userCode = userCode
        self.gameActive = False

        # PGN setup
        self.year = time.strftime("%Y")
        self.month = time.strftime("%m")
        self.day = time.strftime("%d")
        self.pgnGame = chess.pgn.Game()
        self.pgnGame.headers["Event"] = "Lokales Multiplayer Spiel"
        self.pgnGame.headers["Date"] = f"{self.year}.{self.month}.{self.day}"
        self.pgnGame.headers["White"] = self.userCode
        self.pgnGame.headers["Black"] = "Spieler 2"
        self.pgnNode = self.pgnGame
        self.filename = f"{self.userCode}.pgn"

        # Server setup
        self.serverAddress = ("10.5.5.110", 5001)
        self.clientSocket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)

        # NeoPixel setup
        self.pixelPin = board.D12
        self.numPixels = 64
        self.pixels = pixels

        # Matrix setup
        self.rows = rows
        self.cols = cols
        self.tiles = tiles
        self.squares = squares

    def returnToMenu(self):
        self.gameActive = False
        self.pixels.fill(self.colorOff)
        self.pixels.show()
        self.board.reset()

    def calibrateBoard(self):
        for r, row in enumerate(self.rows):
            row.on()
            time.sleep(self.rowChargeTime)
            rank = r + 1
            for c, col in enumerate(self.cols):
                field = self.tiles[c] + str(rank)
                if col.value == 0:
                    self.previousSensor[field] = 0
                else:
                    self.previousSensor[field] = 1
            row.off()
            time.sleep(self.rowDischargeTime)

    def startingAnimation(self):
        self.pixels.fill(self.colorOff)
        self.pixels.show()
        for i in range(self.numPixels):
            self.pixels[i] = self.animationColor
            if i % 8 == 0:
                time.sleep(0.1)
            self.pixels.show()
        time.sleep(1)
        for i in range(self.numPixels):
            self.pixels[i] = self.colorOff
            if i % 8 == 0:

        self.pixels.fill(self.animationColor)
        self.pixels.show()
        time.sleep(0.5)

        self.pixels.fill(self.colorOff)
        if self.pickedSquare is not None:
            self.setSingleLed(self.pickedSquare, self.colorSource)
            for move in self.legalMoves:
                targetSquare = move.to_square
                if self.board.is_capture(move):
                    self.setSingleLed(targetSquare, self.colorCapture)
                else:
                    self.setSingleLed(targetSquare, self.colorTarget)
        if self.pinSquare is not None:
            self.setSingleLed(self.pinSquare, self.colorPinner)

        self.pixels.show()

    def postGame(self, boardTurn):
        # Define field ranges
        whiteSquares = ["A1", "B1", "C1", "D1", "E1", "F1", "G1", "H1",
                        "A2", "B2", "C2", "D2", "E2", "F2", "G2", "H2"]
        blackSquares = ["A7", "B7", "C7", "D7", "E7", "F7", "G7", "H7",
                        "A8", "B8", "C8", "D8", "E8", "F8", "G8", "H8"]
        
        # Blink 5 times
        for _ in range(5):
            self.pixels.fill(self.colorOff)
            self.pixels.show()
            time.sleep(0.3)
            
            # Light up the fields based on boardTurn
            if boardTurn == chess.WHITE:
                # White won: green for white squares, red for black squares
                for square in whiteSquares:
                    if square in self.squares:
                        self.pixels[self.squares[square]] = self.colorRed
                for square in blackSquares:
                    if square in self.squares:
                        self.pixels[self.squares[square]] = self.colorGreen
            else:
                # Black won: red for white squares, green for black squares
                for square in whiteSquares:
                    if square in self.squares:
                        self.pixels[self.squares[square]] = self.colorGreen
                for square in blackSquares:
                    if square in self.squares:
                        self.pixels[self.squares[square]] = self.colorRed

            self.pixels.show()
            time.sleep(0.3)
        try:
            self.clientSocket.connect(self.serverAddress)

            with open(self.filename, "rb") as file:
                data = file.read()
                self.clientSocket.sendall(data)

            self.clientSocket.close()
        except:
            pass
        
        self.gameActive = False

    def savePGN(self):
        with open(self.filename, "w") as f:
            f.write(str(self.pgnGame))

    def boardScanCycle(self):
        """
        Scans the board for changes. Turn on each row, check columns,
        then turn off the row. Detects piece lifts and landings, 
        checks for legal moves and handles checkmate condition.
        """
        for r, row in enumerate(self.rows):
            row.on()
            time.sleep(self.rowChargeTime)

            rank = r + 1
            for c, col in enumerate(self.cols):
                field = self.tiles[c] + str(rank)

                if col.value == 0:
                    currentState = 0
                else:
                    currentState = 1

                if field in self.previousSensor:
                    previous = self.previousSensor[field]
                else:
                    previous = currentState

                # Lift
                if previous == 1 and currentState == 0:
                    if self.pickedSquare is None:
                        sq = chess.parse_square(field.lower())
                        piece = self.board.piece_at(sq)

                        # Only show moves if it's the right color's turn
                        if piece and piece.color == self.board.turn:
                            self.pickedSquare = sq
                            self.legalMoves = []
                            for move in self.board.legal_moves:
                                if move.from_square == sq:
                                    self.legalMoves.append(move)
                            self.updateLedBoard()

                # Land
                elif previous == 0 and currentState == 1:
                    if self.pickedSquare is not None:
                        time.sleep(self.debounceTime)
                        if col.value == 1:
                            sq = chess.parse_square(field.lower())

                            foundMove = None
                            for move in self.legalMoves:
                                if move.to_square == sq:
                                    foundMove = move
                                    break

                            if foundMove:
                                self.board.push(foundMove)

                                # Checkmate check
                                if self.board.is_checkmate():
                                    if self.board.turn == chess.WHITE:
                                        self.postGame(self.board.turn)
                                    else:
                                        self.postGame(self.board.turn)

                                self.pgnNode = self.pgnNode.add_variation(foundMove)
                                self.savePGN()

                                self.pickedSquare = None
                                self.legalMoves = []
                                self.pixels.fill(self.colorOff)
                                self.pixels.show()
                                self.calibrateBoard()
                            elif sq == self.pickedSquare:
                                self.pickedSquare = None
                                self.legalMoves = []
                                self.pixels.fill(self.colorOff)
                                self.pixels.show()

                self.previousSensor[field] = currentState

            row.off()
            time.sleep(self.rowDischargeTime)
            
    def run(self):
        self.gameActive = True
        self.calibrateBoard()
        self.startingAnimation()

        try:
            while self.gameActive:
                self.boardScanCycle()
                time.sleep(0.01)
            self.returnToMenu()
        
        except KeyboardInterrupt:
            try:
                self.clientSocket.connect(self.serverAddress)

                with open(self.filename, "rb") as file:
                    data = file.read()
                    self.clientSocket.sendall(data)

                self.clientSocket.close()
            except:
                pass

            self.pixels.fill(self.colorOff)
            self.pixels.show()

        finally:
            self.returnToMenu()
            os.remove(self.filename)

                        if piece and piece.color == self.chessboard.turn:
                            self.pickedSquare = sq
                            self.legalMoves = []
                            for move in self.board.legal_moves:
                                if move.from_square == sq:
                                    self.legalMoves.append(move)

                            if expectedComMove:
                                self.displayComLed(expectedComMove)
                            else:
                                self.updateLedBoard()

                # If figure is put down, sensor goes from 0 to 1
                elif previous == 0 and currentState == 1:
                    if self.pickedSquare is not None:
                        time.sleep(self.debounceTime)
                        if col.value == 1:
                            sq = chess.parse_square(field.lower())

                            foundMove = None
                            for move in self.legalMoves:
                                if move.to_square == sq:
                                    foundMove = move
                                    break

                            if foundMove:
                                self.board.push(foundMove)
                                self.chessboard.push(foundMove)
                                self.pickedSquare = None
                                self.legalMoves = []
                                self.updateLedBoard()
                                return foundMove
                            elif sq == self.pickedSquare:
                                # Figure was put back on its original square
                                self.pickedSquare = None
                                self.legalMoves = []
                                if expectedComMove:
                                    self.displayComLed(expectedComMove)
                                else:
                                    self.updateLedBoard()

                # Checkmate check
                if self.board.is_checkmate():
                    if self.board.turn == chess.WHITE:
                        self.postGame(self.board.turn)
                    else:
                        self.postGame(self.board.turn)
                    self.returnToMenu()

                self.previousSensor[field] = currentState

            row.off()
            time.sleep(self.rowDischargeTime)
        return detectedMove

    def run(self):
        self.gameActive = True
        self.calibrateBoard()
        self.startingAnimation()

        comMoveTarget = None

        try:
            while self.gameActive:
                # Check if it's COM's turn and if move is not yet performed
                currentTurnIsCom = self.chessboard.turn == chess.BLACK

                if not currentTurnIsCom:
                    comMoveTarget = None

                if (
                    currentTurnIsCom
                    and comMoveTarget is None
                    and not self.chessboard.is_game_over()
                ):
                    limit = chess.engine.Limit(time=0.1)
                    result = self.engine.play(self.chessboard, limit)
                    comMoveTarget = result.move
                    self.displayComLed(comMoveTarget)

                performedMove = self.boardScanCycle(
                                expectedComMove=comMoveTarget)

                # LED Handler for player
                if (
                    self.pickedSquare is None
                    and self.pinSquare is None
                    and not currentTurnIsCom
                ):
                    pass
                elif self.pickedSquare is not None and not currentTurnIsCom:
                    self.updateLedBoard()

                # Handle moves
                if performedMove:
                    if currentTurnIsCom:
                        if performedMove == comMoveTarget:
                            self.pixels.fill(self.colorOff)
                            self.setSingleLed(performedMove.to_square,
                                              self.colorSuccess)
                            self.pixels.show()
                            time.sleep(1.0)
                            self.calibrateBoard()
                        else:
                            self.pixels.fill(self.colorError)
                            self.pixels.show()
                            time.sleep(1.0)
                            # Shows correct move again
                            self.displayComLed(comMoveTarget)

                            # Wrong move is thrown out
                            self.board.pop()
                            self.chessboard.pop()

                    else:
                        self.calibrateBoard()

                time.sleep(0.01)

        except KeyboardInterrupt:
            self.pixels.fill(self.colorOff)
            self.pixels.show()
            self.engine.quit()

        finally:
            self.pixels.fill(self.colorOff)
            self.pixels.show()
            self.engine.quit()
